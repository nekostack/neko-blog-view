                        var info_top = $(".info").offset().top;
                        $(window).scroll(function(){
                            let top = $(document).scrollTop();
                            let width = $(".info").width();
                            let height = $(".info").height();
                            if(top >= info_top){
                                $(".info").addClass("fixed");
                                $(".info").css("width",width);
                                $(".info").css("height",height);
                            }else{
                                $(".info").removeClass("fixed");
                            }
                        })
                    })
